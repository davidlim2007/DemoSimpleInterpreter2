// Set the variable 'a' to value "Hello "
SET a = "Hello "

// Set the variable 'b' to value "World"
SET b = "World"

// Concatenate 'b' to 'a'
CONCAT a, b

// Set the variable 'c' to value 100
SET c = 100

// Multiply 'c' by value 10.
MUL c, 10

// Divide 'c' by value 50.
DIV c, 50

// Print out values of 'a', 'b', 'c'.
GET a
GET b
GET c

EXIT
